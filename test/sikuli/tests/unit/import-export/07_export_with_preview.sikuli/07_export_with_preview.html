import configurator, os

scs_import_str = " SCS Import"

def go_to_current_dir():
    scs_base = configurator.get_path_property("SCSBasePath")
    find(Pattern().exact()).left().paste(Pattern().exact(), scs_base); type(Key.ENTER)

p = configurator.start_it_up(getBundlePath(), "startup.blend")
try:
    wait(Pattern().similar(0.90), 5); hover(Pattern().similar(0.90)); type(Key.ESC)
    click(Pattern().similar(0.90))
    find(Pattern().similar(0.90)); hover(Pattern().similar(0.90))
    type(Key.ESC); find(Pattern().similar(0.90))
    click(Pattern().similar(0.90))
    find(Pattern().similar(0.90)); hover(Pattern().similar(0.90))

    type(Key.ENTER);
    wait(0.5); type(Key.ENTER); mouseMove(Location(0,0)); hover(Pattern().similar(0.90))
    type("3"); click();
    type(scs_import_str + Key.ENTER); go_to_current_dir()
    click(Pattern().similar(0.90)); type(Key.ENTER); find(Pattern().similar(0.90))
    type("4");
    type(scs_import_str + Key.ENTER); go_to_current_dir()
    click(Pattern().similar(0.90)); type(Key.ENTER); find(Pattern().similar(0.90))
except:
    configurator.save_screenshot(getBundlePath(), Screen())
    raise
finally:
    configurator.close_blender(p)
